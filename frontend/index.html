async function findMatches() { const name = document.getElementById("name").value.trim(); const grade = parseInt(document.getElementById("grade").value); const subject = document.getElementById("subject").value; if (!name) { alert("Please enter the mentee's name"); return; } currentMentee = { name, grade, subject }; const btn = document.getElementById("find-btn"); btn.disabled = true; btn.textContent = "SEARCHING..."; document.getElementById("results-card").style.display = "block"; document.getElementById("results").innerHTML = `
Finding best matches using RAG + your trained model...
`; document.getElementById("status-msg").innerHTML = ""; try { const res = await fetch(`${API}/match`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ mentee_name: name, mentee_grade: grade, mentee_subject: subject }), }); const data = await res.json(); renderMatches(data.matches); } catch (err) { document.getElementById("results").innerHTML = `
Could not connect to backend. Is the server running?
`; } btn.disabled = false; btn.textContent = "FIND MATCHES →"; } function renderMatches(matches) { const el = document.getElementById("results"); el.innerHTML = matches.map((m, i) => `
${ScoreRing(m.match_score)}
${m.name} ${i === 0 ? 'BEST MATCH' : ''}
Grade ${m.grade} · ${m.subject} · ${m.qualifications}
${m.explanation}
`).join(""); } async function book(idx, btn) { if (!currentMentee) return; const mentor = JSON.parse(btn.dataset.mentor); btn.textContent = "..."; btn.disabled = true; try { await fetch(`${API}/book`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ mentor_name: mentor.name, mentee_name: currentMentee.name, subject: currentMentee.subject, mentor_grade: mentor.grade, mentee_grade: currentMentee.grade, match_score: mentor.match_score, explanation: mentor.explanation, }), }); btn.textContent = "✓ BOOKED"; btn.classList.add("booked"); document.getElementById("status-msg").innerHTML = `
Pairing saved: ${currentMentee.name} → ${mentor.name}
`; } catch { btn.textContent = "BOOK"; btn.disabled = false; } }
